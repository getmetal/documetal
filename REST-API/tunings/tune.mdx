---
title: "Add Tuning"
api: "POST https://api.getmetal.io/v1/tune"
description: "Tune your embeddings for better results."
---

### Auth Headers

<ParamField required header="x-metal-api-key" type="string">
  An API key for your org.
</ParamField>

<ParamField required header="x-metal-client-id" type="string">
  A Client ID for your org.
</ParamField>

### Body

<ParamField body="index" type="string" placeholder="Index ID" required>
  Index ID
</ParamField>

<ParamField body="idA" type="string" placeholder="Embedding ID A" required>
  Document ID A
</ParamField>

<ParamField body="idB" type="string" placeholder="Embedding ID B" required>
  Document ID B
</ParamField>

<ParamField
  body="label"
  type="number"
  placeholder="Score result. Either -1, 0, or 1"
  required
>
  Closeness score. Either `-1`, `0`, or `1`
</ParamField>

### Response

<ResponseField name="data" type="Tuning Object">
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Id of the Tuning
    </ResponseField>

    <ResponseField name="index" type="string">
      Id of the index for the Tuning
    </ResponseField>

    <ResponseField name="idA" type="string">
     Document ID A
    </ResponseField>

    <ResponseField name="idB" type="string">
      Document ID B
    </ResponseField>

    <ResponseField name="label" type="string">
      The score label. Either `-1`, `0`, or `1`
    </ResponseField>

  </Expandable>
</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request POST 'https://api.getmetal.io/v1/tune' \
--header 'Content-Type: application/json' \
--header 'x-metal-api-key: <api-key>' \
--header 'x-metal-client-id: <client-id>' \
--data-raw '{
    "index": "index-id",
    "idA": "embedding-1",
    "idB": "embedding-2",
    "label": -1
}'
```
