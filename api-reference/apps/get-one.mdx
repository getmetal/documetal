---
title: "Get App"
api: "GET https://api.getmetal.io/v1/apps/{appId}"
description: "This endpoint gets a single index."
---

### Cookie Header

<ParamField required header="token"  type="string">
  This is the users auth token.
</ParamField>

<ParamField required path="appId" type="string">
  The App ID
</ParamField>

### Response

<ResponseField name="data" type="App Object">
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Id of the app
    </ResponseField>

    <ResponseField name="name" type="string">
      Name of the app
    </ResponseField>

    <ResponseField name="activeIndexId" type="string">
      Id of the active index
    </ResponseField>
  </Expandable>
</ResponseField>
